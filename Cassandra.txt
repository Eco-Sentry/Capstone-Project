CREATE TABLE sensorreading (
    id UUID,
    sensorType UUID,
    sensorId UUID,
    dateTime TIMESTAMP,
    reading DOUBLE,
    latitude DOUBLE,
    longitude DOUBLE,
    trust DOUBLE,
    PRIMARY KEY ((sensorType), dateTime, id)
) WITH CLUSTERING ORDER BY (dateTime DESC, id ASC);
